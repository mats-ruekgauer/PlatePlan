# Dish List Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a Recipes tab with two sub-tabs — Verlauf (history of every cooked dish with week label) and Meine Rezepte (manually entered personal recipes fed into plan generation).

**Architecture:** A `source` column on the `recipes` table distinguishes AI-generated from manual recipes. A new `useCookedMeals` hook queries `planned_meals` joined to `meal_plans` and `recipes`, filtering for `cooked`/`prepared` status. The existing `recipes.tsx` tab gets a two-tab switcher and is unhidden in the bottom navigation. A new `app/recipe/new.tsx` screen provides a full form for manual recipe entry. The `generate-plan` Edge Function is updated to include manual recipes in the Claude prompt.

**Tech Stack:** React Native, NativeWind, Expo Router, React Query (@tanstack/react-query), Supabase (PostgREST client), Deno Edge Functions, TypeScript.

```ts
import { useQuery } from '@tanstack/react-query';

import { getCurrentUserId, mapRecipe, supabase } from '../lib/supabase';
import type { MealSlot, Recipe } from '../types';

// ─── Types ────────────────────────────────────────────────────────────────────

export interface CookedMealEntry {
  recipe: Recipe;
  weekStart: string;
  mealSlot: MealSlot;
  plannedMealId: string;
}

// ─── Query key ────────────────────────────────────────────────────────────────

export const cookedMealsKey = ['cooked-meals'] as const;

// ─── Week label helper ────────────────────────────────────────────────────────

/**
 * Converts an ISO week-start date (always a Monday) to a human-readable
 * ISO week label, e.g. "KW 15 · 2026".
 */
export function toWeekLabel(weekStart: string): string {
  // Parse as UTC noon to avoid DST / timezone edge cases
  const d = new Date(weekStart + 'T12:00:00Z');
  const dayNum = d.getUTCDay() || 7; // 1=Mon … 7=Sun
  d.setUTCDate(d.getUTCDate() + 4 - dayNum); // shift to Thursday of the same ISO week
  const yearStart = new Date(Date.UTC(d.getUTCFullYear(), 0, 1));
  const weekNo = Math.ceil(((d.getTime() - yearStart.getTime()) / 86400000 + 1) / 7);
  return `KW ${weekNo} · ${d.getUTCFullYear()}`;
}

// ─── Hook ─────────────────────────────────────────────────────────────────────

/**
 * Returns every meal the user has cooked or prepared, most recent first.
 * Each cooked instance is its own entry (no deduplication).
 */
export function useCookedMeals() {
  return useQuery({
    queryKey: cookedMealsKey,
    queryFn: async (): Promise<CookedMealEntry[]> => {
      const userId = await getCurrentUserId();

      // 1. Fetch all of this user's plans to get plan IDs and week_start values
      const { data: plans, error: plansError } = await supabase
        .from('meal_plans')
        .select('id, week_start')
        .eq('user_id', userId);
      if (plansError) throw plansError;
      if (!plans || plans.length === 0) return [];

      const planIds = plans.map((p) => p.id);
      const planWeekMap: Record<string, string> = Object.fromEntries(
        plans.map((p) => [p.id, p.week_start]),
      );

      // 2. Fetch cooked planned_meals joined with both possible recipe references
      const { data: rows, error } = await supabase
        .from('planned_meals')
        .select(`
          id,
          plan_id,
          meal_slot,
          chosen_recipe_id,
          recipe:recipes!planned_meals_recipe_id_fkey (*),
          chosenRecipe:recipes!planned_meals_chosen_recipe_id_fkey (*)
        `)
        .in('plan_id', planIds)
        .in('status', ['cooked', 'prepared'])
        .order('created_at', { ascending: false });
      if (error) throw error;

      return (rows ?? []).map((row) => {
        // eslint-disable-next-line @typescript-eslint/no-explicit-any
        const r = row as any;
        const resolvedRecipe: Recipe =
          row.chosen_recipe_id && r.chosenRecipe
            ? mapRecipe(r.chosenRecipe)
            : mapRecipe(r.recipe);

        return {
          recipe: resolvedRecipe,
          weekStart: planWeekMap[row.plan_id] ?? '',
          mealSlot: row.meal_slot as MealSlot,
          plannedMealId: row.id,
        };
      });
    },
    staleTime: 1000 * 60 * 5,
  });
}
```

- [ ] **Step 1: Create the form screen**

```tsx
import { router } from 'expo-router';
import React, { useState } from 'react';
import {
  Alert,
  KeyboardAvoidingView,
  Platform,
  ScrollView,
  Text,
  TextInput,
  TouchableOpacity,
  View,
} from 'react-native';
import { useQueryClient } from '@tanstack/react-query';

import { getCurrentUserId, supabase } from '../../lib/supabase';

// ─── Types ────────────────────────────────────────────────────────────────────

interface IngredientRow {
  name: string;
  amount: string;
  unit: string;
}

// ─── Screen ───────────────────────────────────────────────────────────────────

export default function NewRecipeScreen() {
  const queryClient = useQueryClient();

  const [title, setTitle] = useState('');
  const [description, setDescription] = useState('');
  const [cuisine, setCuisine] = useState('');
  const [cookTime, setCookTime] = useState('');
  const [servings, setServings] = useState('1');
  const [ingredients, setIngredients] = useState<IngredientRow[]>([
    { name: '', amount: '', unit: '' },
  ]);
  const [steps, setSteps] = useState<string[]>(['']);
  const [tags, setTags] = useState('');
  const [kcal, setKcal] = useState('');
  const [protein, setProtein] = useState('');
  const [carbs, setCarbs] = useState('');
  const [fat, setFat] = useState('');
  const [saving, setSaving] = useState(false);

  // ─── Ingredient helpers ─────────────────────────────────────────────────────

  function updateIngredient(index: number, field: keyof IngredientRow, value: string) {
    setIngredients((prev) => prev.map((ing, i) => (i === index ? { ...ing, [field]: value } : ing)));
  }

  function addIngredient() {
    setIngredients((prev) => [...prev, { name: '', amount: '', unit: '' }]);
  }

  function removeIngredient(index: number) {
    setIngredients((prev) => prev.filter((_, i) => i !== index));
  }

  // ─── Step helpers ────────────────────────────────────────────────────────────

  function updateStep(index: number, value: string) {
    setSteps((prev) => prev.map((s, i) => (i === index ? value : s)));
  }

  function addStep() {
    setSteps((prev) => [...prev, '']);
  }

  function removeStep(index: number) {
    setSteps((prev) => prev.filter((_, i) => i !== index));
  }

  // ─── Save ────────────────────────────────────────────────────────────────────

  async function handleSave() {
    if (!title.trim()) {
      Alert.alert('Titel fehlt', 'Bitte gib einen Titel für das Rezept ein.');
      return;
    }

    setSaving(true);
    try {
      const userId = await getCurrentUserId();

      const filteredIngredients = ingredients
        .filter((i) => i.name.trim().length > 0)
        .map((i) => ({
          name: i.name.trim(),
          amount: parseFloat(i.amount) || 0,
          unit: i.unit.trim() || 'Stück',
        }));

      const filteredSteps = steps.map((s) => s.trim()).filter((s) => s.length > 0);

      const parsedTags = tags
        .split(',')
        .map((t) => t.trim())
        .filter(Boolean);

      const { error } = await supabase.from('recipes').insert({
        user_id: userId,
        title: title.trim(),
        description: description.trim() || null,
        cuisine: cuisine.trim() || null,
        cook_time_minutes: cookTime ? parseInt(cookTime, 10) : null,
        servings: parseInt(servings, 10) || 1,
        ingredients: filteredIngredients,
        steps: filteredSteps,
        tags: parsedTags,
        calories_per_serving: kcal ? parseInt(kcal, 10) : null,
        protein_per_serving_g: protein ? parseInt(protein, 10) : null,
        carbs_per_serving_g: carbs ? parseInt(carbs, 10) : null,
        fat_per_serving_g: fat ? parseInt(fat, 10) : null,
        is_seasonal: false,
        season: 'all',
        estimated_price_eur: null,
        source: 'manual',
      });

      if (error) throw error;

      queryClient.invalidateQueries({ queryKey: ['recipes', 'manual'] });
      router.back();
    } catch (err) {
      Alert.alert('Fehler', err instanceof Error ? err.message : 'Rezept konnte nicht gespeichert werden.');
    } finally {
      setSaving(false);
    }
  }

  // ─── Render ──────────────────────────────────────────────────────────────────

  return (
    <KeyboardAvoidingView
      className="flex-1 bg-[#F8F9FA]"
      behavior={Platform.OS === 'ios' ? 'padding' : undefined}
    >
      {/* Header */}
      <View className="flex-row items-center justify-between px-4 pt-14 pb-3">
        <TouchableOpacity
          onPress={() => router.back()}
          className="w-9 h-9 rounded-xl bg-white items-center justify-center active:opacity-70"
          style={{ shadowColor: '#000', shadowOpacity: 0.07, shadowRadius: 3, shadowOffset: { width: 0, height: 1 } }}
        >
          <Text className="text-lg">‹</Text>
        </TouchableOpacity>

        <Text className="text-lg font-bold text-[#1A1A2E]">Neues Rezept</Text>

        <TouchableOpacity
          onPress={handleSave}
          disabled={saving}
          className="px-4 py-2 rounded-xl bg-[#2D6A4F] active:opacity-70"
        >
          <Text className="text-sm font-semibold text-white">
            {saving ? 'Speichert…' : 'Speichern'}
          </Text>
        </TouchableOpacity>
      </View>

      <ScrollView
        className="flex-1"
        contentContainerClassName="px-4 pb-12 gap-5"
        keyboardShouldPersistTaps="handled"
      >
        {/* Basic info */}
        <FormSection title="Grunddaten">
          <FormField label="Titel *">
            <TextInput
              className="bg-white rounded-xl px-4 py-3 text-[#1A1A2E] border border-gray-200"
              placeholder="z.B. Spaghetti Bolognese"
              value={title}
              onChangeText={setTitle}
            />
          </FormField>

          <FormField label="Beschreibung">
            <TextInput
              className="bg-white rounded-xl px-4 py-3 text-[#1A1A2E] border border-gray-200"
              placeholder="Kurze Beschreibung (optional)"
              value={description}
              onChangeText={setDescription}
              multiline
              numberOfLines={2}
            />
          </FormField>

          <View className="flex-row gap-3">
            <View className="flex-1">
              <FormField label="Küche">
                <TextInput
                  className="bg-white rounded-xl px-4 py-3 text-[#1A1A2E] border border-gray-200"
                  placeholder="Italienisch"
                  value={cuisine}
                  onChangeText={setCuisine}
                />
              </FormField>
            </View>
            <View className="flex-1">
              <FormField label="Kochzeit (min)">
                <TextInput
                  className="bg-white rounded-xl px-4 py-3 text-[#1A1A2E] border border-gray-200"
                  placeholder="30"
                  value={cookTime}
                  onChangeText={setCookTime}
                  keyboardType="numeric"
                />
              </FormField>
            </View>
          </View>

          <View className="flex-row gap-3">
            <View className="flex-1">
              <FormField label="Portionen">
                <TextInput
                  className="bg-white rounded-xl px-4 py-3 text-[#1A1A2E] border border-gray-200"
                  placeholder="1"
                  value={servings}
                  onChangeText={setServings}
                  keyboardType="numeric"
                />
              </FormField>
            </View>
            <View className="flex-1">
              <FormField label="Tags (Komma)">
                <TextInput
                  className="bg-white rounded-xl px-4 py-3 text-[#1A1A2E] border border-gray-200"
                  placeholder="vegan, schnell"
                  value={tags}
                  onChangeText={setTags}
                />
              </FormField>
            </View>
          </View>
        </FormSection>

        {/* Macros */}
        <FormSection title="Nährwerte (pro Portion, optional)">
          <View className="flex-row gap-3">
            <View className="flex-1">
              <FormField label="kcal">
                <TextInput
                  className="bg-white rounded-xl px-4 py-3 text-[#1A1A2E] border border-gray-200"
                  placeholder="450"
                  value={kcal}
                  onChangeText={setKcal}
                  keyboardType="numeric"
                />
              </FormField>
            </View>
            <View className="flex-1">
              <FormField label="Protein (g)">
                <TextInput
                  className="bg-white rounded-xl px-4 py-3 text-[#1A1A2E] border border-gray-200"
                  placeholder="30"
                  value={protein}
                  onChangeText={setProtein}
                  keyboardType="numeric"
                />
              </FormField>
            </View>
          </View>
          <View className="flex-row gap-3">
            <View className="flex-1">
              <FormField label="Kohlenhydrate (g)">
                <TextInput
                  className="bg-white rounded-xl px-4 py-3 text-[#1A1A2E] border border-gray-200"
                  placeholder="50"
                  value={carbs}
                  onChangeText={setCarbs}
                  keyboardType="numeric"
                />
              </FormField>
            </View>
            <View className="flex-1">
              <FormField label="Fett (g)">
                <TextInput
                  className="bg-white rounded-xl px-4 py-3 text-[#1A1A2E] border border-gray-200"
                  placeholder="15"
                  value={fat}
                  onChangeText={setFat}
                  keyboardType="numeric"
                />
              </FormField>
            </View>
          </View>
        </FormSection>

        {/* Ingredients */}
        <FormSection title="Zutaten">
          {ingredients.map((ing, idx) => (
            <View key={idx} className="flex-row gap-2 items-center">
              <View className="flex-1">
                <TextInput
                  className="bg-white rounded-xl px-3 py-3 text-[#1A1A2E] border border-gray-200"
                  placeholder="Zutat"
                  value={ing.name}
                  onChangeText={(v) => updateIngredient(idx, 'name', v)}
                />
              </View>
              <View style={{ width: 60 }}>
                <TextInput
                  className="bg-white rounded-xl px-3 py-3 text-[#1A1A2E] border border-gray-200 text-center"
                  placeholder="100"
                  value={ing.amount}
                  onChangeText={(v) => updateIngredient(idx, 'amount', v)}
                  keyboardType="numeric"
                />
              </View>
              <View style={{ width: 60 }}>
                <TextInput
                  className="bg-white rounded-xl px-3 py-3 text-[#1A1A2E] border border-gray-200 text-center"
                  placeholder="g"
                  value={ing.unit}
                  onChangeText={(v) => updateIngredient(idx, 'unit', v)}
                />
              </View>
              {ingredients.length > 1 && (
                <TouchableOpacity
                  onPress={() => removeIngredient(idx)}
                  className="w-8 h-8 rounded-full bg-red-100 items-center justify-center"
                >
                  <Text className="text-red-500 text-lg font-bold" style={{ lineHeight: 20 }}>−</Text>
                </TouchableOpacity>
              )}
            </View>
          ))}
          <TouchableOpacity
            onPress={addIngredient}
            className="flex-row items-center gap-2 py-1"
          >
            <View className="w-7 h-7 rounded-full bg-[#D8F3DC] items-center justify-center">
              <Text className="text-[#2D6A4F] font-bold text-base" style={{ lineHeight: 20 }}>+</Text>
            </View>
            <Text className="text-sm text-[#2D6A4F] font-medium">Zutat hinzufügen</Text>
          </TouchableOpacity>
        </FormSection>

        {/* Steps */}
        <FormSection title="Zubereitung">
          {steps.map((step, idx) => (
            <View key={idx} className="flex-row gap-2 items-start">
              <View className="w-7 h-7 rounded-full bg-[#2D6A4F] items-center justify-center flex-shrink-0 mt-2">
                <Text className="text-xs font-bold text-white">{idx + 1}</Text>
              </View>
              <View className="flex-1">
                <TextInput
                  className="bg-white rounded-xl px-3 py-3 text-[#1A1A2E] border border-gray-200"
                  placeholder={`Schritt ${idx + 1}`}
                  value={step}
                  onChangeText={(v) => updateStep(idx, v)}

                />
              </View>
              {steps.length > 1 && (

                  <Text className="text-red-500 text-lg font-bold" style={{ lineHeight: 20 }}>−</Text>
                </TouchableOpacity>
              )}
            </View>
          ))}
          <TouchableOpacity
            onPress={addStep}
            className="flex-row items-center gap-2 py-1"
          >
            <View className="w-7 h-7 rounded-full bg-[#D8F3DC] items-center justify-center">
              <Text className="text-[#2D6A4F] font-bold text-base" style={{ lineHeight: 20 }}>+</Text>
            </View>
            <Text className="text-sm text-[#2D6A4F] font-medium">Schritt hinzufügen</Text>
          </TouchableOpacity>
        </FormSection>

    </KeyboardAvoidingView>
  );
```
