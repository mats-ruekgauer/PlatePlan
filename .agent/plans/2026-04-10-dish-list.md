## Purpose / Big Picture

After this change, users can:
1. Open a "Recipes" tab in the bottom navigation and browse every dish they have actually cooked, with an ISO week label (e.g. "KW 15 · 2026") showing when it was cooked.
2. Add their own recipes manually (title, description, ingredients, steps, macros, etc.) via a form screen.
3. Have those manually-added recipes automatically included in the next AI meal plan generation.

Currently the Recipes tab exists as a file but is hidden (`href: null`) and there is no way to add personal recipes.

## Context and Orientation

**Stack:** React Native + Expo (SDK 51), NativeWind 4 (Tailwind for RN), Expo Router (file-based), React Query v5, Supabase (PostgREST JS client), TypeScript, Deno Edge Functions (Supabase).

**Key files and their roles:**
- `types/index.ts` — all shared TypeScript types; `Recipe` interface lives here
- `lib/supabase.ts` — Supabase client singleton, `Database` type (manually maintained), and all row mapper functions (`mapRecipe`, `mapMealPlan`, etc.)
- `hooks/usePlan.ts` — reference for how Supabase joins with multiple FK references to the same table work in this project (see `recipes!planned_meals_recipe_id_fkey`)
- `hooks/useFavorites.ts` — reference for a typical hook pattern with React Query
- `app/(tabs)/recipes.tsx` — existing screen, currently shows all recipes grouped by cuisine, but is **hidden** in `app/(tabs)/_layout.tsx` via `href: null`
- `app/(tabs)/_layout.tsx` — bottom tab navigator; recipes tab must be unhidden here
- `app/recipe/[id].tsx` — detail screen; already works and needs no changes
- `supabase/functions/generate-plan/index.ts` — Deno Edge Function; calls Claude to generate meal plans
- `supabase/functions/_shared/prompts.ts` — shared prompt builders for the Edge Functions
- `supabase/migrations/` — SQL migration files, numbered 001–005 (two files named 005 exist — use 006 for new migration)

**Data model relevant to this feature:**
- `recipes` table: stores all recipes with `user_id` (null = global/shared, non-null = user-specific). After migration, also has `source` column.
- `planned_meals` table: one row per meal slot in a plan. Has `recipe_id`, `alternative_recipe_id`, `chosen_recipe_id` (null = use `recipe_id`), `status`, and `plan_id`.
- `meal_plans` table: one row per weekly plan. Has `user_id` and `week_start` (always a Monday ISO date).
- `planned_meals` does NOT have a `user_id` column — user filtering goes via `plan_id → meal_plans.user_id`.

**FK aliases used in Supabase joins (from `usePlan.ts`):**
```
recipe:recipes!planned_meals_recipe_id_fkey (*)
chosenRecipe:recipes!planned_meals_chosen_recipe_id_fkey (*)
```
Use these exact aliases in `useCookedMeals`.

**Conventions:**
- snake_case in DB, camelCase in app — always go through mapper functions
- All Supabase queries in hooks use `getCurrentUserId()` from `lib/supabase.ts`
- React Query `staleTime: 1000 * 60 * 5` is standard across all hooks
- NativeWind classes for styling; primary green: `#2D6A4F`, dark text: `#1A1A2E`, muted text: `#6B7280`
- Back navigation uses `router.back()` from `expo-router`
- Alert for user-facing errors uses React Native's `Alert.alert`

### Milestone 3: useCookedMeals Hook

Step 1: Create hook file

    // File: hooks/useCookedMeals.ts

    import { useQuery } from '@tanstack/react-query';

    import { getCurrentUserId, mapRecipe, supabase } from '../lib/supabase';
    import type { MealSlot, Recipe } from '../types';

    // ─── Types ────────────────────────────────────────────────────────────────────

    export interface CookedMealEntry {
      recipe: Recipe;
      weekStart: string;
      mealSlot: MealSlot;
      plannedMealId: string;
    }

    // ─── Query key ────────────────────────────────────────────────────────────────

    export const cookedMealsKey = ['cooked-meals'] as const;

    // ─── Week label helper ────────────────────────────────────────────────────────

    /**
     * Converts an ISO week-start date (always a Monday) to a human-readable
     * ISO week label, e.g. "KW 15 · 2026".
     * Uses noon UTC to avoid DST / timezone edge cases.
     */
    export function toWeekLabel(weekStart: string): string {
      const d = new Date(weekStart + 'T12:00:00Z');
      const dayNum = d.getUTCDay() || 7; // 1=Mon … 7=Sun
      d.setUTCDate(d.getUTCDate() + 4 - dayNum); // shift to Thursday of the same ISO week
      const yearStart = new Date(Date.UTC(d.getUTCFullYear(), 0, 1));
      const weekNo = Math.ceil(((d.getTime() - yearStart.getTime()) / 86400000 + 1) / 7);
      return `KW ${weekNo} · ${d.getUTCFullYear()}`;
    }

    // ─── Hook ─────────────────────────────────────────────────────────────────────

    /**
     * Returns every meal the user has cooked or prepared, most recent first.
     * Each cooked instance is its own entry (no deduplication).
     */
    export function useCookedMeals() {
      return useQuery({
        queryKey: cookedMealsKey,
        queryFn: async (): Promise<CookedMealEntry[]> => {
          const userId = await getCurrentUserId();

          // 1. Get all plan IDs + week_start values for this user
          const { data: plans, error: plansError } = await supabase
            .from('meal_plans')
            .select('id, week_start')
            .eq('user_id', userId);
          if (plansError) throw plansError;
          if (!plans || plans.length === 0) return [];

          const planIds = plans.map((p) => p.id);
          const planWeekMap: Record<string, string> = Object.fromEntries(
            plans.map((p) => [p.id, p.week_start]),
          );

          // 2. Fetch cooked planned_meals with both recipe FK references
          const { data: rows, error } = await supabase
            .from('planned_meals')
            .select(`
              id,
              plan_id,
              meal_slot,
              chosen_recipe_id,
              recipe:recipes!planned_meals_recipe_id_fkey (*),
              chosenRecipe:recipes!planned_meals_chosen_recipe_id_fkey (*)
            `)
            .in('plan_id', planIds)
            .in('status', ['cooked', 'prepared'])
            .order('created_at', { ascending: false });
          if (error) throw error;

          return (rows ?? []).map((row) => {
            // eslint-disable-next-line @typescript-eslint/no-explicit-any
            const r = row as any;
            const resolvedRecipe: Recipe =
              row.chosen_recipe_id && r.chosenRecipe
                ? mapRecipe(r.chosenRecipe)
                : mapRecipe(r.recipe);

            return {
              recipe: resolvedRecipe,
              weekStart: planWeekMap[row.plan_id] ?? '',
              mealSlot: row.meal_slot as MealSlot,
              plannedMealId: row.id,
            };
          });
        },
        staleTime: 1000 * 60 * 5,
      });
    }

      const [carbs, setCarbs] = useState('');
      const [fat, setFat] = useState('');
      const [saving, setSaving] = useState(false);

      function updateIngredient(idx: number, field: keyof IngredientRow, value: string) {
        setIngredients((prev) => prev.map((ing, i) => (i === idx ? { ...ing, [field]: value } : ing)));
      }
      function addIngredient() { setIngredients((prev) => [...prev, { name: '', amount: '', unit: '' }]); }
      function removeIngredient(idx: number) { setIngredients((prev) => prev.filter((_, i) => i !== idx)); }

      function removeStep(idx: number) { setSteps((prev) => prev.filter((_, i) => i !== idx)); }
